sun_from_north.1 = {
	type = character_event
	title = sun_from_north.1.t
	desc = sun_from_north.1.desc
	theme = hunting
	override_background = {
		event_background = sitting_room
	}
	left_portrait = root

	trigger = {
		NOT = { has_global_variable = nk_rising }
		has_title = title:e_north_korea
	}
	immediate = {
		set_global_variable = nk_rising
		root = { save_scope_as = avgvstvs }
	}
	option = {
		name = sun_from_north.1.a
		spawn_army = {
			uses_supply = no
			inheritable = yes
			name = Korean_People's_Army1
			levies = {
  				value = 0
				every_held_title = {
					limit = {
 						tier = tier_county
					}
					add = 1000
				}
			}
			location = root.capital_province
		}
		spawn_army = {
			uses_supply = no
			inheritable = yes
			name = Korean_People's_Army2
			levies = {
  				value = 0
				every_held_title = {
					limit = {
 						tier = tier_county
					}
					add = 1000
				}
			}
			location = root.capital_province
		}
		spawn_army = {
			uses_supply = no
			inheritable = yes
			name = Korean_People's_Army3
			levies = {
  				value = 0
				every_held_title = {
					limit = {
 						tier = tier_county
					}
					add = 1000
				}
			}
			location = root.capital_province
		}
		every_independent_ruler = {
			limit = {
				NOT = { character = root }
			}
			trigger_event = {
				id = sun_from_north.2
				days = 1
			}
		}
		every_player = {
			trigger_event = {
				id = sun_from_north.3
				days = 2
			}
		}
	}
}
sun_from_north.2 = {
	type = letter_event
	opening = {
		desc = sun_from_north.2.t
	}
	sender = scope:avgvstvs
	desc = sun_from_north.2.desc
	option = {
		trigger = {
			OR = {
				root.faith = faith:juche
				root.faith = faith:haeseoan
			}
		}
		name = sun_from_north.2.a
		create_title_and_vassal_change = {
			type = swear_fealty
			save_scope_as = change
			add_claim_on_loss = no
		}
		root = {
			change_liege = {
				liege = scope:avgvstvs
				change = scope:change
			}
			resolve_title_and_vassal_change = scope:change
		}
	}
	option = {
		trigger = {
			root.faith = faith:inmin_juche
			is_ai = no
		}
		name = sun_from_north.2.b
		create_title_and_vassal_change = {
			type = swear_fealty
			save_scope_as = change
			add_claim_on_loss = no
		}
		root = {
			change_liege = {
				liege = scope:avgvstvs
				change = scope:change

	}
}
