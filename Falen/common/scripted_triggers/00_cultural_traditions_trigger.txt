# Overlapping Tradition Triggers

# For checking if a CulTrad can be picked when you already have one of its parameters from elsewhere.
lacks_overlapping_cultrad_trigger = {
	custom_description = {
		text = lacks_overlapping_cultrad_tt
		NOT = {
			any_in_list = {
				list = traits
				this = flag:$TRADITION$
			}
		}
	}
}
