rebuild_gangwon_decision_1 = {
	picture = {
		reference = "gfx/interface/illustrations/decisions/decision_dynasty_house.dds"
	}
	decision_group_type = major
	ai_check_interval = 60
	desc = rebuild_gangwon_decision_1_desc

	is_shown = {
		has_global_variable = gw_rising
		has_variable = gw_emperor
		global_var:gw_rising == 0
	}

	is_valid = {
		culture = { has_cultural_pillar = heritage_gwandong }
		is_independent_ruler = yes

		completely_controls = title:d_chuncheon
		completely_controls = title:d_wonju
	}
	
	is_valid_showing_failures_only = {
		is_capable_adult = yes
		is_imprisoned = no
		is_at_war = no
	}
	
	effect = {
		trigger_event = {
			id = gw_mgga.11
		}
		hidden_effect = {
			title:d_pyeongchang.holder = {
				save_scope_as = pyeongchang
				trigger_event = {
					id = gw_mgga.3
					delayed = yes
				}
			}
			create_title_and_vassal_change = {
				type = created
				save_scope_as = change
				add_claim_on_loss = no
			}
			title:k_yeongseo = {
				change_title_holder = {
					holder = root
					change = scope:change
				}
			}
			resolve_title_and_vassal_change = scope:change
			
			change_global_variable = {
				name = gw_rising
				add = 1
			}
		}
	
	}
	
	ai_potential = {
		always = yes
	}
	
	ai_will_do = {
		base = 100
	}
}
