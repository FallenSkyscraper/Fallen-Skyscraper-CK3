run_for_presidency_decision = {
	picture = "gfx/interface/illustrations/decisions/decision_invite_knights.dds"
	ai_check_interval = 10


	desc = run_for_presidency_decision_desc
	selection_tooltip = run_for_presidency_decision_tooltip

	is_shown = {
		is_ai = yes
		is_imprisoned = no
		OR = {
			AND = {
				is_ruler = yes
				is_landed = yes
				is_independent = no
				top_liege = {
					has_title = title:k_prk
				}
			}
			is_primary_heir_of = title:k_prk
			has_title = title:k_prk
		}
		NOT = { has_variable = candidate }
	}
	is_valid = {
		prestige_level >= 1
		NOT = {
			any_living_character = {
				has_variable = candidate
				count > 8
			}
		}
	}
	cost = { gold = 60 prestige = 100 }
	effect = {
		set_variable = candidate
	}
	ai_potential = {
		always = yes
	}
	ai_will_do = {
		base = 5

		modifier = {
			add = {
				value = ai_boldness
				multiply = 0.2
			}
		}
		modifier = {
			add = {
				value = ai_energy
				multiply = 0.5
			}
		}
		modifier = {
			add = 200
			has_trait = ambitious
		}
		modifier = {
			add = -100
			has_trait = content
		}
		modifier = {
			add = ai_sociability
		}
		modifier = {
			add = {
				value = ai_honor
				multiply = -0.2
			}
		}
		modifier = {
			add = 200
			has_title = title:k_prk
		}
	}
}
delay_election_decision = {
	picture = "gfx/interface/illustrations/decisions/decision_invite_knights.dds"
	ai_check_interval = 10

	desc = delay_election_decision_desc
	
	selection_tooltip = delay_election_decision_tooltip

	is_shown = {
		is_imprisoned = no
		has_title = title:k_prk
	}
	is_valid = {
		prestige_level >= 1
	}
	cost = { prestige = 400 }
	effect = {
		change_global_variable = { name = election_date add = 4 }
		change_global_variable = { name = election_term add = 4 }
		custom_tooltip = delay_election_decision_tt
		if = {
			limit = { is_at_war = yes }
			every_vassal = {
				add_opinion = {
					modifier = dictactor_opinion
					opinion = -10 # 전시면 좀 봐줌
					target = prev
				}
			}
		}
		else = {
			custom_tooltip = delay_election_decision_tt2
			hidden_effect = { add_prestige_level = -1 }
			every_vassal = {
				add_opinion = {
					modifier = dictactor_opinion
					target = prev
				}
			}
		}
	}
	ai_potential = {
		always = yes
	}
	ai_will_do = {
		base = 0
	}
}
