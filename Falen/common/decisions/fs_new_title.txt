form_gangwon_decision = {
	picture = "gfx/interface/illustrations/decisions/decision_dynasty_house.dds"
	major = yes
	ai_check_interval = 60
	desc = form_gangwon_decision_desc

	is_shown = {
		culture = { has_cultural_pillar = heritage_gangwon }
		NOT = {
			is_target_in_global_variable_list = {
				name = unavailable_unique_decisions
				target = flag:form_gangwon_decision
			}
		}
	}

	is_valid = {
		culture = { has_cultural_pillar = heritage_gwandong }
		is_independent_ruler = yes

		highest_held_title_tier =< tier_kingdom

		AND = {
			completely_controls = title:k_cherwon
			completely_controls = title:k_yeongseo
			completely_controls = title:k_yeongdong
		}
	}

	effect = {
		save_scope_as = gangwon_former

		show_as_tooltip = { form_gangwon_decision_effects = yes }

		#Events
		trigger_event = new_title.0001
		every_player = {
			limit = {
				NOT = { this = scope:gangwon_former }
				is_within_diplo_range = { CHARACTER = scope:gangwon_former }
			}
			trigger_event = new_title.0002
		}
		every_vassal_or_below = {
			limit = {
				is_ai = yes
				exists = capital_province
				capital_province = {
					county = {
						any_this_title_or_de_jure_above = {
							OR = {
								this = title:k_cherwon
								this = title:k_yeongseo
								this = title:k_yeongdong
							}
						}
					}
				}
			}
			trigger_event = new_title.0002 #So they can convert
		}

		#Can only happen once
		add_to_global_variable_list = {
			name = unavailable_unique_decisions
			target = flag:form_gangwon_decision
		}
		set_global_variable = {
			name = form_gangwon_decision
			value = scope:gangwon_former
		}
	}

	cost = { gold = { value = 300 } }

	ai_potential = {
		# Only for the AI if independent ruler
		is_independent_ruler = yes
	}

	ai_will_do = {
		base = 100
	}
}
