#Scripted Modifiers used for Scripted Relations (friends, rivals, lovers etc.)

#Will range from factor 1 (at high positive opinion for both) to factor 2 (at low positive opinion for both)
best_friend_event_weight_modifier = {
	#Positive_high or higher
	modifier = {
		any_relation = {
			type = best_friend
			opinion = {
				target = root
				value >= high_positive_opinion
			}
		}
		factor = 1
	}
	modifier = {
		any_relation = {
			type = best_friend
			reverse_opinion = {
				target = root
				value >= high_positive_opinion
			}
		}
		factor = 1
	}
	#Between positive_high and positive_medium
	modifier = {
		any_relation = {
			type = best_friend
			opinion = {
				target = root
				value < high_positive_opinion
				value >= medium_positive_opinion
			}
		}
		factor = 1.15
	}
	modifier = {
		any_relation = {
			type = best_friend
			reverse_opinion = {
				target = root
				value < high_positive_opinion
				value >= medium_positive_opinion
			}
		}
		factor = 1.15
	}
	#Between positive_medium and positive_low
	modifier = {
		any_relation = {
			type = best_friend
			opinion = {
				target = root
				value < medium_positive_opinion
				value > low_positive_opinion
			}
		}
		factor = 1.3
	}
	modifier = {
		any_relation = {
			type = best_friend
			reverse_opinion = {
				target = root
				value < medium_positive_opinion
				value > low_positive_opinion
			}
		}
		factor = 1.3
	}
	#positive_low or lower
	modifier = {
		any_relation = {
			type = best_friend
			opinion = {
				target = root
				value <= low_positive_opinion
			}
		}
		factor = 1.4
	}
	modifier = {
		any_relation = {
			type = best_friend
			reverse_opinion = {
				target = root
				value <= low_positive_opinion
			}
		}
		factor = 1.4
	}
}
