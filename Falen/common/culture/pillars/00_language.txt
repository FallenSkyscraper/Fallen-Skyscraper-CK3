language_anglic = {
	type = language
	is_shown = {
		language_is_shown_trigger = {
			LANGUAGE = language_anglic
		}
	}
	ai_will_do = {
		value = 10
		if = {
			limit = { has_cultural_pillar = language_anglic }
			multiply = 10
		}
	}
	
	color = english
}

language_arabic = {
	type = language
	is_shown = {
		language_is_shown_trigger = {
			LANGUAGE = language_arabic
		}
	}
	ai_will_do = {
		value = 10
		if = {
			limit = { has_cultural_pillar = language_arabic }
			multiply = 10
		}
		if = {
			limit = {
				scope:character = {
					religion = religion:islam_religion
				}
			}
			multiply = 2
		}
	}
	
	color = bedouin
}
