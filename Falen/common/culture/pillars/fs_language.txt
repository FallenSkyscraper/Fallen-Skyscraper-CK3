## Koreanic Languages
language_old_korean = {
	type = language
	is_shown = {
		language_is_shown_trigger = {
			LANGUAGE = language_old_korean
		}
	}
	parameters = {
		#language_group_korean = yes
	}
	ai_will_do = {
		value = 10
		if = {
			limit = { has_cultural_pillar = language_old_korean }
			multiply = 10
		}
	}
	
	color = white
}

	type = language
	is_shown = {
		language_is_shown_trigger = {
			LANGUAGE = language_arabic
		}
	}
	ai_will_do = {
		value = 10
		if = {
			limit = { has_cultural_pillar = language_arabic }
			multiply = 10
		}
		if = {
			limit = {
				scope:character = {
					religion = religion:islam_religion
				}
			}
			multiply = 2
		}
	}
	
	color = bedouin
