hec_gui_toggle = {
	scope = character
	
	is_shown = {
		has_variable = hec_gui_toggle
	}

	effect = {
		if = {
			limit = {
				has_variable = hec_gui_toggle
			}
			remove_variable = hec_gui_toggle
		}
		else = {
			set_variable = hec_gui_toggle
		}
	}
}

hec_gui_close = {
	scope = character
	
	is_shown = {
		always = yes
	}

	effect = {
		remove_variable = hec_gui_toggle
	}
}
hec_gui_abc = {
	scope = character
	is_shown = {
		root = {
			OR = {
				has_title = title:e_hec
				top_liege = { has_title = title:e_hec }
			}
		}
		#always = no
	}
}
hec_get_emperor = {
	scope = character
	effect = {
		title:e_hec = {
			holder = {
				root = {
					set_variable = {
						name = hec_emperor
						 value = prev
					}
				}
			}
		}
	}
}

reform_approve = {
	scope = character
	is_valid = {
		OR = {
			has_trait = hec_reformist
			AND = {
				has_trait = hec_religious
				title:e_hec.holder.opinion = {
					target = title:k_imperial_church.holder
					value >= 50
				}
			}
		}
	}
}
reform_against = {
	scope = character
	is_valid = {
		OR = {
			has_trait = hec_conservative

				has_trait = hec_religious
				title:e_hec.holder.opinion = {
					target = title:k_imperial_church.holder

				}

		}
	}
