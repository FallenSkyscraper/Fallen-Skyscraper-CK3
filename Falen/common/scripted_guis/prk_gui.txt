prk_gui_toggle = {
	scope = character
	
	is_shown = {
		has_variable = prk_gui_toggle
	}

	effect = {
		if = {
			limit = {
				has_variable = prk_gui_toggle
			}
			remove_variable = prk_gui_toggle
		}
		else = {
			set_variable = prk_gui_toggle
		}
	}
}

prk_gui_close = {
	scope = character
	
	is_shown = {
		always = yes
	}

	effect = {
		remove_variable = prk_gui_toggle
	}
}
prk_gui_abc = {
	scope = character
	is_shown = {
		root = {
			OR = {
				has_title = title:k_prk
				top_liege = { has_title = title:k_prk }
			}
		}
	}
}
prk_get_president = {
	scope = character
	effect = {
		title:k_prk = {
			holder = {
				root = {
					set_variable = {
						name = prk_president
						 value = prev
					}
				}
			}
		}
	}
}
prk_candidate = {
	scope = character
	effect = {
		every_living_character = {
			limit = { has_variable = candidate }
			root = {
				add_to_variable_list = {
					name = prk_candidates
					target = prev
				}
			}
		}
	}
}
prk_clear_list = {
	scope = character
	effect = {
		clear_variable_list = prk_candidates
	}
}
prk_run_candidate = {
	scope = character
	is_valid = {
		root = {
			NOT = { has_variable = candidate }
			gold >= 80
			prestige >= 100
			prestige_level >= 1
			OR = {
				has_title = title:k_prk
				top_liege = { has_title = title:k_prk }
			}
		}
	}
	effect = {
		root = {
			set_variable = candidate
			remove_short_term_gold = 100
			add_prestige = -100
		}
	}
}

prk_resign_candidate = {
	scope = character
	is_valid = {
		root = { has_variable = candidate }
	}
	effect = {
		root = {
			remove_variable = candidate
		}
	}
}
election_gui_toggle = {
	scope = character
	
	is_shown = {
		has_variable = election_gui_toggle
	}
}
voting1 = {
	scope = character
	effect = {
		root = {
			set_variable = { name = voting_power value = 0 }
			every_held_title = {
				limit = {
					tier = tier_county
					development_level > 0
				}
				prev = {
					change_variable = { name = voting_power add = prev.development_level }
				}
			}
		}
	}
}
voting = {
	scope = character
	is_valid = { always = yes }
	effect = {
		scope:player = {
			root = {
				change_variable = { name = votes add = prev.var:voting_power }
			}
			remove_variable = voting_power
			remove_variable = election_gui_toggle
		}
		title:k_prk.holder = {
			trigger_event = {
				id = joson_election.3
				days = 1
			}
		}
	}
}
voting_absent = {
	scope = character
	effect = {
		remove_variable = election_gui_toggle
		remove_variable = voting_power
		title:k_prk.holder = {
			trigger_event = {
				id = joson_election.3
				days = 1
			}
		}
	}
}
