
dungeon_events_on_action_chance = {
	if = {
		limit = {
			exists = imprisoner
			scope:imprisoner = { has_character_flag = reformed_dungeon }
		}
		add = 25
	}
	if = {
		limit = {
			exists = scope:imprisoner
			scope:imprisoner = { has_character_flag = neglected_dungeon }
		}
		add = 75
	}
	else = {
		add = 50
	}
}
