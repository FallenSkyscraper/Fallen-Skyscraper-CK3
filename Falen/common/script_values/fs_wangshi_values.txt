wangshi_diplomacy_task_scale = {
    add = diplomacy
    if = {
        limit = {
            scope:councillor_liege = { has_perk = loyalty_and_respect_perk }
        }
        multiply = 1.25
    }
    divide = 3
    if = {
        limit = {
            scope:councillor_liege = { has_perk = loyalty_and_respect_perk }
        }
        ceiling = yes
    }
    else = {
        floor = yes
    }
    min = 1
}
