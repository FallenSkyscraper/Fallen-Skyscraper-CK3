	}
}

spouse_martial_event_troop_size = {
	add = martial
	add = prowess
	if = {
		limit = {
			has_trait = strategist
		}
		multiply = 1.5
	}
	if = {
		limit = {
			has_trait = overseer
		}
		multiply = 1.5
	}
	if = {
		limit = {
			has_trait = gallant
		}
		multiply = 1.5
	}

	multiply = 10

	round = yes
}

	add = 1

	multiply = scope:councillor.martial

	if = {
		limit = {
			scope:councillor = { has_trait = strategist }
		}
		add = 1
	}
	else_if = {
		limit = {
			scope:councillor = { has_trait = overseer }
		}
		add = 2
	}
	if = {
		limit = {
			scope:councillor = { has_trait = gallant }
		}
		add = 3
	}

	min = 8
	max = 25
}
