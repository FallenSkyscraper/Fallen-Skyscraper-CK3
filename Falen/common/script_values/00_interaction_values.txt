golden_obligation_value = {
	value = medium_gold_value
}

ransom_cost_value = {
	if = {
		limit = {
			exists = imprisoner
			imprisoner = { has_perk = golden_obligations_perk }
			NOT = {
				exists = liege
			}
		}
		value = {
			add = ransom_cost
			add = 90
			min = imprisoner.golden_obligation_value
		}
	}
	else_if = {
		limit = {
			NOT = {
				exists = liege
			}
		}
		value = { # Rich guests can pay a bit more...
			add = ransom_cost
			if = {
				limit = {
					gold >= 100
				}
				add = 90
			}
		}
	}
	else_if = {
		limit = {
			exists = imprisoner
			imprisoner = { has_perk = golden_obligations_perk }
			ransom_cost < imprisoner.golden_obligation_value
		}
		value = imprisoner.golden_obligation_value
	}
	else = {
		value = ransom_cost
	}
}

half_ransom_cost_value = {
	value = {
		add = ransom_cost_value
		divide = 2
		min = 1
	}
}

grant_title_opinion_sum_baronies = {
	if = { # if no other titles present, calc for baronies
		limit = {
			any_in_list = {
				list = titles_to_grant
				tier = tier_barony
			}
		}

		every_in_list = {
			list = titles_to_grant
			limit = {
				tier = tier_barony
			}
			add = 20 # barony grant opinion
		}
	}
}
