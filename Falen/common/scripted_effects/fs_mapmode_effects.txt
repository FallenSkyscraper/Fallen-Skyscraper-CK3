tech_value_processing = {
	every_barony = {
		if = {
			limit = {
				title_capital_county = { var:tech_$TECH$ > 9.0 }
			}
			set_color_from_title = title:d_color_gradient_11
		}
		else_if = {
			limit = {
				title_capital_county = { var:tech_$TECH$ > 8.0 }
			}
			set_color_from_title = title:d_color_gradient_10
