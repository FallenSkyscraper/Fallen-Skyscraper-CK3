spawn_kor_character_effect = {
    title:c_ulleung.title_province = {
	    save_scope_as = $INVADER_DYNN$_spawn_location
	}
	
	create_character = {
	    location = scope:$INVADER_DYNN$_spawn_location
		template = kor_rise_character_template
		sace_scope_as = $INVADER_NAME$
	}
	
	scope:$INVADER_NAME$ = {
	    add_gold = 2000
		add_prestige = 2500
		add_piety = 1000
		
		dynasty = {
		    set_dynasty_name = test
			set_coa = title:e_korea
			
			add_dynasty_prestige_level = 3
			add_dynasty_prestige = 10000
		}
		house = {
		    set_house_name = hi_dynn_test
		}
	}
	
	every_player = {
	    trigger_event = kor_rise.0001
	}
}

		INVADER_DYNN = $INVADER_DYNN$
		INVADER_CHAR = $INVADER_CHAR$

		INVADER_CHAR = $INVADER_CHAR$

	spawn_invader_servants_effect = {
		INVADER_CHAR = $INVADER_CHAR$
